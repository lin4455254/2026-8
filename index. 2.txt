    <script>
        const CAMP_WEEKS = [
            { id: 'week1', label: '第一週 (8/03 - 8/07)' },
            { id: 'week2', label: '第二週 (8/10 - 8/14)' },
            { id: 'week3', label: '第三週 (8/17 - 8/21)' },
        ];
        const PRICING = { 0: 0, 1: 7000, 2: 12000, 3: 15000 };

        let state = {
            weeks: [],
            participants: [{ name: '', age: '' }],
            total: 0,
            individual: 0
        };

        function initWeeks() {
            const list = document.getElementById('weeks-list');
            list.innerHTML = '';
            CAMP_WEEKS.forEach(w => {
                const div = document.createElement('div');
                div.id = `item-${w.id}`;
                div.className = "week-item flex items-center justify-between p-5 rounded-2xl border-2 cursor-pointer border-[#f0ecec] bg-[#fcfafa] transition-all";
                div.innerHTML = `<span class="font-bold text-[#b4b4b4]">${w.label}</span><div class="check w-6 h-6 rounded-lg border-2 border-[#d1d1d1] bg-white"></div>`;
                div.addEventListener('click', () => toggleWeek(w.id));
                list.appendChild(div);
            });
        }

        function toggleWeek(id) {
            const idx = state.weeks.indexOf(id);
            if (idx > -1) state.weeks.splice(idx, 1);
            else state.weeks.push(id);
            const el = document.getElementById(`item-${id}`);
            if (state.weeks.includes(id)) el.classList.add('active');
            else el.classList.remove('active');
            updateTotal();
        }

        function changeCount(n) {
            document.querySelectorAll('.count-btn').forEach(b => {
                b.classList.remove('bg-white', 'text-[#b8a6a6]', 'shadow-sm');
                b.classList.add('text-[#b4b4b4]');
                if (parseInt(b.dataset.val) === n) {
                    b.classList.add('bg-white', 'text-[#b8a6a6]', 'shadow-sm');
                    b.classList.remove('text-[#b4b4b4]');
                }
            });
            const old = [...state.participants];
            state.participants = Array.from({length: n}, (_, i) => old[i] || {name: '', age: ''});
            renderParticipants();
            updateTotal();
        }

        function renderParticipants() {
            const container = document.getElementById('participants-container');
            container.innerHTML = '';
            state.participants.forEach((p, i) => {
                const div = document.createElement('div');
                div.className = "grid grid-cols-3 gap-4 p-5 bg-[#fcfafa] rounded-[1.5rem] border border-[#f0ecec]";
                div.innerHTML = `
                    <input required placeholder="學員姓名" class="col-span-2 px-4 py-3 text-sm rounded-xl border border-[#e2d5d5] outline-none focus:border-[#b8a6a6] bg-white" value="${p.name}">
                    <input required type="number" placeholder="年齡" class="px-4 py-3 text-sm rounded-xl border border-[#e2d5d5] outline-none focus:border-[#b8a6a6] bg-white" value="${p.age}">
                `;
                const inputs = div.querySelectorAll('input');
                inputs[0].addEventListener('input', (e) => { state.participants[i].name = e.target.value; });
                inputs[1].addEventListener('input', (e) => { state.participants[i].age = e.target.value; });
                container.appendChild(div);
            });
        }

        function updateTotal() {
            state.individual = PRICING[state.weeks.length] || 0;
            let base = state.individual * state.participants.length;
            let note = `共 ${state.participants.length} 位`;
            if (state.participants.length === 2) { base *= 0.95; note += " (二人95折)"; }
            else if (state.participants.length >= 3) { base *= 0.9; note += " (三人9折)"; }
            state.total = Math.round(base);
            document.getElementById('individual-display').innerText = `$${state.individual.toLocaleString()}`;
            document.getElementById('total-display').innerText = `$${state.total.toLocaleString()}`;
            document.getElementById('participant-summary').innerText = note;
        }

        document.getElementById('copy-btn').addEventListener('click', () => {
            const text = document.getElementById('account-num').innerText;
            const el = document.createElement('textarea');
            el.value = text; document.body.appendChild(el);
            el.select(); document.execCommand('copy');
            document.body.removeChild(el);
            alert("匯款帳號已複製到剪貼簿");
        });

        document.querySelectorAll('.count-btn').forEach(btn => {
            btn.addEventListener('click', () => changeCount(parseInt(btn.dataset.val)));
        });

        document.getElementById('enrollment-form').addEventListener('submit', async function(e) {
            e.preventDefault();
            if (state.weeks.length === 0) return alert("請至少選擇一個週數");
            
            const submitBtn = document.getElementById('submit-btn');
            submitBtn.disabled = true;
            submitBtn.innerText = "提交中...";

            const parentName = document.getElementById('parentName').value;
            const phone = document.getElementById('phone').value;
            const bankLastFive = document.getElementById('bankLastFive').value;

            // 組合 Line 訊息
            const weekStr = CAMP_WEEKS.filter(w => state.weeks.includes(w.id)).map(w => w.label.split(' ')[0]).join('、');
            const pStr = state.participants.map((p, idx) => `${idx + 1}. ${p.name} (${p.age}歲)`).join('\n');
            const discountLabel = state.participants.length === 3 ? " (三人9折)" : (state.participants.length === 2 ? " (二人95折)" : "");
            
            const msg = `您好！我已完成網頁報名預約 ✨\n\n👤 家長：${parentName}\n📞 電話：${phone}\n📅 報名週數：${weekStr}\n👧 學員名單：\n${pStr}\n💰 應繳總額：NT$ ${state.total.toLocaleString()}${discountLabel}\n🔢 帳號後五碼：${bankLastFive || '待匯款後提供'}\n\n請協助確認名額，謝謝！`;
            
            const lineUrl = `https://line.me/R/oaMessage/@904sjunh/?${encodeURIComponent(msg)}`;

            // 1. 切換到第二頁顯示
            document.getElementById('form-container').classList.add('hidden');
            document.getElementById('success-container').classList.remove('hidden');
            document.getElementById('success-total').innerText = `NT$ ${state.total.toLocaleString()}`;
            document.getElementById('line-link-btn').href = lineUrl;
            window.scrollTo(0, 0);

            // 2. 背景存檔 (Firebase)
            try {
                if (typeof __firebase_config !== 'undefined' && __firebase_config) {
                    const conf = JSON.parse(__firebase_config);
                    if (!firebase.apps.length) firebase.initializeApp(conf);
                    const db = firebase.firestore();
                    await firebase.auth().signInAnonymously();
                    await db.collection('artifacts').doc('rhythmic-gymnastics-2026').collection('public').doc('data').collection('registrations').add({
                        parentName, phone, bankLastFive, weeks: state.weeks, participants: state.participants, 
                        totalPrice: state.total, submittedAt: new Date().toISOString()
                    });
                }
            } catch (err) {
                console.error("Firebase update failed", err);
            }
        });

        window.onload = () => {
            initWeeks();
            renderParticipants();
            updateTotal();
        };
    </script>
</body>
